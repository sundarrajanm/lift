    private var cachedGreeting: String? = null

    fun greet(): String {
        if (cachedGreeting == null) {
            val message = "Hello from Smart Commit Assistant!"
            println("👋 Greeting user: $message")
            cachedGreeting = message
        } else {
            println("♻️ Using cached greeting")
        }
        return cachedGreeting!!
    }
